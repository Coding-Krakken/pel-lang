QUICK START COMMANDS
================================================================================

# Run marketing analysis
./pel compile marketing_cac_ltv.pel -o marketing.ir.json
./pel run marketing.ir.json --mode deterministic --horizon 24 -o results.json
python3 view_marketing.py results.json

# Check product development constraints
./pel compile product_development.pel -o product.ir.json
./pel run product.ir.json --mode deterministic --horizon 18 -o results.json
cat results.json | jq '.constraint_violations | length'

# Analyze historical data
python3 analyze_historical_data.py

# Run risk scenarios
./pel compile risk_scenarios.pel -o risk.ir.json
./pel run risk.ir.json --mode deterministic --horizon 18 -o results.json

# Personal business workflow
./run_my_analysis.sh

# View complete portfolio
python3 showcase_portfolio.py
